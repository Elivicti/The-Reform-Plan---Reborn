#改革计划
ap_tt_reform_plane = {
	on_enabled = {
		custom_tooltip = "tt_reform_plane_desc"
		hidden_effect = {
			country_event = { id = tt_reform_plane.4 }
			set_country_flag = flag_the_reform_plane
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_3"
			num_ascension_perks > 2
		}
	}
	potential = {
		NOT = {
			has_ascension_perk = ap_tt_reform_plane
		}
	}
	modifier = {
		country_base_unity_produces_add = 10
		country_base_influence_produces_add = 5
		country_minor_artifacts_produces_mult = 1
	}
	ai_weight = {
		factor = 1
		modifier = {
			factor = 2
			years_passed >= 100
		}
		modifier = {
			factor = 2
			mid_game_years_passed >= 0
		}
		modifier = {
			factor = 2
			end_game_years_passed >= 0
		}
	}
}
